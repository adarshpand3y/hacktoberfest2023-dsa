//brute approach
#include<bits/stdc++.h>
bool isPalindrome(string str){
	int n= str.length();
	int i=0,j = n-1;
	
	while(i<j){
		if(str[i]!=str[j]){
			return false;
		}
		i++,j--;
	}
	return true;
}
int minCharsPalindrome(string str){
	int count =0;
	while(str.length()>0){
		if(isPalindrome(str)){
			break;
		}
		else{
			count++;
			str.erase(str.begin()+str.length()-1);
		}
	}
	return count;
}

int minCharsforPalindrome(string str) {

	int start=0;

	while(start<end){
		if(str[start]==str[end]){
			start++;
			end--;
		}
		else{
			res++;
			start=0;
			end=str.length()-res-1;
		}
	}
	return res;
}

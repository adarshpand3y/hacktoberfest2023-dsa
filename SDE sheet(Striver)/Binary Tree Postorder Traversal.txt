//using binary tree + recursive
#include<bits/stdc++.h>
void postorder(TreeNode*root,vector<int>&vec){
	if(root==NULL){
		return;
	}
	postorder(root->left,vec);
	postorder(root->right,vec);
	vec.push_back(root->val);

vector<int>postorderTraversal(TreeNode*root){
	vector<int>vec;
	postorder(root,vec);
	return vec;
}

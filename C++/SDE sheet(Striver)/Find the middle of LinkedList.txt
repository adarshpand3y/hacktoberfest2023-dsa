//naive approach
ListNode*middleNode(ListNode*head){
int n=0;
ListNode*temp = head;
while(temp){
	n++;
	temp = temp->next;
}
temp = head;
for(int i=0;i<n/2;i++){
	temp = temp->next;
}
return temp;
}

//two pointer approach
ListNode*middleNode(ListNode*head){
	ListNode*slow = head;
	ListNode*fast = head;
	
	while(fast && fast->next){
		fast= fast->next->next;
		slow = slow->next;
		return slow;
	}
}

string commonPrefix(vector<string>&arr,int n){

	string ans = arr[0];
        for (int i = 1; i < n; i++) {
          int j;
          for (j = 0; j < min(arr[i].size(), ans.size()); j++)
            if(ans[j] != arr[i][j])
              break;
          ans = ans.substr(0, j);
        }
        return (ans.size() == 0 ? "-1" : ans);
}

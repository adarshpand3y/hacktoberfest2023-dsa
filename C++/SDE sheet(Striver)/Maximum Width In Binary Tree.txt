//using level order traversal
#include<bits/stdc++.h>
int getMaxWidth(TreeNode<int>*root)
{
    // Write your code here.
    queue<TreeNode<int>*>q;
    q.push(root);
    q.push(NULL);
    int maxCnt = INT_MIN;
    int cnt =0;
    while(!q.empty()){
        auto it = q.front();
        q.pop();
        if(it==NULL){
            maxCnt = max(maxCnt,cnt);
            cnt =0;
            if(q.front()!=NULL)
            q.push(NULL);
            continue;
        }
        else
        cnt++;
        if(it->left!=NULL)
        q.push(it->left);
        if(it->right!=NULL)
        q.push(it->right);
    }
    return maxCnt;
}

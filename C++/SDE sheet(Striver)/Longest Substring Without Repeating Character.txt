//Optimised appraoch
#include<bits/stdc++.h>
using namespace std;
int lengthOfLongestSubstring(string s){
	vector<int>freq(256,-1);
	int left =0,right = 0;
	int n = s.size();
	int len =0;
	while(right<n){
		if(freq[s[right]]!=-1)
		left = max(freq[s[right]]+1,left);
		
		freq[s[right]] = right;
		len = max(len,right-left+1);
		right++;
	}
	return len;
}
